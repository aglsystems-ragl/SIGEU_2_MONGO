db.evaluacion_evento.aggregate([
  {
    $lookup: {
      from: "usuario",
      localField: "usuario_id",
      foreignField: "_id",
      as: "evaluador"
    }
  },
  { $unwind: "$evaluador" },
  {
    $lookup: {
      from: "evento",
      localField: "evento_id",
      foreignField: "_id",
      as: "evento"
    }
  },
  { $unwind: "$evento" }
]);
